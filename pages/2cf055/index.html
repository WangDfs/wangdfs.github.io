        int x = grid.length;
        int y = grid[0].length;

        int[][] dp = new int[x + 1][y + 1];
        dp[0][0] = grid[0][0];
        for(int i = 1; i < y; i++){
            dp[0][i] = grid[0][i] + dp[0][i - 1];

            dp[i][0] = grid[i][0] + dp[i - 1][0];

            }
        }
        return dp[x - 1][y - 1];
    }

            cnt1[s1.charAt(i) - 'a']++;
            cnt2[s2.charAt(i) - 'a']++;
        }
        if(Arrays.equals(cnt1,cnt2)){

        for(int i = len1; i < len2; i++){
            ++cnt2[s2.charAt(i)- 'a'];
            --cnt2[s2.charAt(i - len1) - 'a'];
            if(Arrays.equals(cnt1,cnt2))return true;
        }
        return false;
    }

    class Solution {
    class Node{
        int num;
        int index;
        public Node(int num, int index){
            this.num = num;
            this.index = index;
        }
    }
    public String frequencySort(String s) {
        PriorityQueue<Node> pq = new PriorityQueue<>(new CompLen());
        int[] nums = new int[150];
        for(int i = 0; i < s.length(); i++){
            nums[s.charAt(i) - '0']++;
        }
        for(int i = 0; i < 150; i++){
            pq.add(new Node(nums[i],i));
        }
        String res = "";
        while(pq.peek().num != 0){
            Node node = pq.poll();
            char c =  (char) (node.index + '0');
            int num = node.num;
            for(int i = 0; i < num; i++){
                res += c;
            }
        }
        return res;
    }
    class CompLen implements Comparator<Node>{
    public int compare(Node o1, Node o2) {
		return o2.num - o1.num; // 也可以实现降序排列 o2.length() - o1.length();
	}
  }
}

    class Solution {
    public int[] nextGreaterElements(int[] nums) {
        
        Stack<Integer> stack = new Stack();

        stack.push(0);
        int[] res = new int[nums.length];
        for(int i = 0; i < res.length; i++){
            res[i] = -1;
        }
        //int j = 0;
        for(int i = 1; i < nums.length * 2 - 1; i++){
            while(!stack.empty() && nums[i % nums.length] > nums[stack.peek()]){
                int index = stack.pop();
                res[index] = nums[i % nums.length];
           }
           stack.push(i % nums.length); 
        }
        return res;
    }
}

    # 2357.使数组中所有元素都等于零
    class Solution {
    public int minimumOperations(int[] nums) {

        PriorityQueue<Integer> pq = new PriorityQueue<>();

        for(int i = 0; i < nums.length; i++){
            if(nums[i] != 0){
                pq.add(nums[i]);
            }
        }
        int cnt = 0;
        int res = 0;
        while(pq.size() > 0){
            int x = pq.poll();
            x -= cnt;
            cnt += x;
            if(x > 0)
            res++;
        }
        return res;
    }
}
