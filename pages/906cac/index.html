并查集

P3367 【模板】并查集 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn) (opens new window)

#include <bits/stdc++.h>
using namespace std;

int n, m;
int f[10010];

int find(int x) {
  if (x == f[x]) {
    return x;
  }
  return f[x] = find(f[x]);
}

void uni(int x, int y) {
  int fx = find(x);
  int fy = find(y);
  if (fx == fy) {
    return;
  }
  f[fx] = fy;
}

int main() {
  scanf("%d%d", &n, &m);
  for (int i = 1; i <= n; i++) {
    f[i] = i;
  }
  for (int i = 1; i <= m; i++) {
    int z, x, y;
    cin >> z >> x >> y;
    if (z == 1) {
      uni(x, y);
    } else {
      if (find(x) == find(y)) {
        cout << "Y" << endl;
      } else {
        cout << "N" << endl;
      }
    }
  }

  return 0;
}
