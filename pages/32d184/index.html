    6201. 找出前缀异或的原始数组 - 力扣（LeetCode） (opens new window)

    要求res[i]，其中pref[i]^res[i+1]=pref[i+1]，两边同时^pref[i]得：res[i+1]=pref[i+1]^pref[i]

    class Solution {
    public int[] findArray(int[] pref) {
      int n = pref.length;
        int[] res = new int[n];
        res[0] = pref[0];
        for (int i = 1; i < n; i++) {
            res[i] = pref[i - 1] ^ pref[i];
        }
        return res;
    }
    
}

    /**
暴力解法，按位来求
*/
class Solution {
    public int[] findArray(int[] pref) {
        int[] res = new int[pref.length];
        res[0] = pref[0];
        for(int i = 1; i < pref.length; i++){
            int t1 = pref[i - 1];
            int t2 = pref[i];
            int[] b1 = new int[32];
            int[] b2 = new int[32];
            for(int j = 0; j < 32; j++){
                b1[j] = t1 >> j & 1;
            }
            for(int j = 0; j < 32; j++){
                b2[j] = t2 >> j & 1;
            }
            int cur = 0;
            for(int j = 0; j < 32; j++){
                int bit = b1[j];
                if(b2[j] != 0){
                    if(bit == 0){
                        bit = 1;
                    }
                    else{
                        bit = 0;
                    }
                }
                if(bit == 1){
                    cur += (bit << j);
                }
            }
            res[i] = cur; 
        }
        return res;
    }
    
}

    6202. 使用机器人打印字典序最小的字符串 - 力扣（LeetCode） (opens new window)

    class Solution {
    public String robotWithString(String s) {
        int[] cnt = new int[26];
        String str = "";
        StringBuilder sb = new StringBuilder();
        for(int i = 0; i < s.length(); i++){
            int x = (int)(s.charAt(i) - 'a');
            cnt[x]++;
        }
        int[] st = new int[s.length()];
        int index = -1;
        int cur = 0;
        for(int i = 0; i < s.length();i++){
           int x = (int)(s.charAt(i) - 'a');
           st[++index] = x;
           cnt[x]--;
           for(int j = cur; j < 26; j++){
               if(cnt[j] > 0){
                   cur = j;
                   break;
               } 
           }
           while(index >= 0 && st[index] <= cur){
               char ss = (char)('a' + st[index]);
                sb.append(ss);
               //str += ss;
               index--;
           } 
        }
        while(index >= 0){
             char ss = (char)('a' + st[index]);
             sb.append(ss);
            // str += ss;
             index--;
        }
        return sb.toString();
    }
}

    提示

    6203. 矩阵中和能被 K 整除的路径 - 力扣（LeetCode） (opens new window)

    class Solution {
    public int numberOfPaths(int[][] grid, int k) {
        int m = grid.length;
        int n = grid[0].length;
        int[][][] map = new int[m][n][k];
        for(int i = 0; i < m; i++){
            for(int j = 0; j < n; j++){
                for(int l = 0; l < k; l++){
                    map[i][j][l] = -1;
                }
            }
        }
        //HashMap<String, Integer> hashMap = new HashMap<>();
        return dfs(grid, m - 1, n - 1, grid[m - 1][n - 1] % k, k, map);
    }

    public int dfs(int[][] grid, int x, int y, int sum, int k,int[][][] map){
        sum = sum % k;
        //String s = "_" + x + "_" + y + "_" + sum;
        if(map[x][y][sum] != -1){
           return  map[x][y][sum];
        }
        // if(hashMap.containsKey(s)){
        //     return hashMap.get(s);
        // }
        if(x < 0 || y < 0)return 0;
        if(x == 0 && y == 0){
            if(sum % k == 0)return 1;
            else{
                return 0;
            }
        }
        int res1 = 0;
        int res2 = 0;
        if(x - 1 >= 0)
        res1 = dfs(grid, x - 1, y,(sum + grid[x - 1][y]) % k, k, map);
        if(y - 1 >= 0)
        res2 = dfs(grid, x,  y - 1, (sum + grid[x][y - 1]) % k, k, map);
       // hashMap.put(s, (res1 + res2) % 1000000007);
        map[x][y][sum] = (res1 + res2) % 1000000007;
        return (res1 + res2) % 1000000007;
    }
}
