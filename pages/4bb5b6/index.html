采药（01背包）

辰辰是个天资聪颖的孩子，他的梦想是成为世界上最伟大的医师。为此，他想拜附近最有威望的医师为师。医师为了判断他的资质，给他出了一个难题。医师把他带到一个到处都是草药的山洞里对他说：“孩子，这个山洞里有一些不同的草药，采每一株都需要一些时间，每一株也有它自身的价值。我会给你一段时间，在这段时间里，你可以采到一些草药。如果你是一个聪明的孩子，你应该可以让采到的草药的总价值最大。”

# 输入格式

第一行有 22 个整数 T（1≤T≤10001≤T≤1000）和 M（1≤M≤1001≤M≤100），用一个空格隔开，T 代表总共能够用来采药的时间，M 代表山洞里的草药的数目。

接下来的 M 行每行包括两个在 11 到 100100 之间（包括 11 和 100100）的整数，分别表示采摘某株草药的时间和这株草药的价值。

# 输出格式

输出在规定的时间内可以采到的草药的最大总价值。

# 输入输出样例

输入 #1

70 3
71 100
69 1
1 2
1
2
3
4

输出 #1

3
1

可以用搜索的方式来求解01背包问题， slove(i, v) 表示当前是第i种草药，当前还剩v体力值

#include <bits/stdc++.h>
using namespace std;
int v[1100], w[1100], f[1100];
int dp[1100][1100];
int t, n;
int dfs(int i, int val) {
  if (dp[i][val] != -1) {
    return dp[i][val];
  }
  if (i > n) {
    return dp[i][val] = 0;
  }
  int res1 = 0;
  if (val - v[i] >= 0) {
    res1 = dfs(i + 1, val - v[i]) + w[i];
  }
  int res2 = dfs(i + 1, val);
  return dp[i][val] = max(res1, res2);
}

int main() {
  cin >> t >> n;
  for (int i = 0; i < 1100; i++)
    for (int j = 0; j < 1100; j++) 
        dp[i][j] = -1; //初始化记忆数组为-1
  for (int i = 1; i <= n; i++) {
    cin >> v[i] >> w[i];
  }
  cout << dfs(1, t);
  return 0;
}

可以用动态规划来求解背包问题

**空间优化前 **

#include <bits/stdc++.h>

#include <iostream>
using namespace std;

int v[1100], w[1100], f[1100][1100];

int main() {
  int t, n;
  cin >> t >> n;

  for (int i = 1; i <= n; i++) {
    cin >> v[i] >> w[i];
  }

  for (int i = 1; i <= n; i++) {
    for (int j = 0; j <= t; j++) {
      f[i][j] = f[i - 1][j];
      if (j >= v[i]) {
        f[i][j] = max(f[i - 1][j], f[i - 1][j - v[i]] + w[i]);
      }
    }
  }

  cout << f[n][t];
  return 0;
}
