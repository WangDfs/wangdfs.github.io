    遍历二叉树(静态数组方式)

    /*
        遍历二叉树, 输入n组关系
        第i组关系表示第i个节点的左孩子是几号节点，右孩子是几号节点。
        为0表示没有节点
*/

#include <algorithm>
#include <iostream>

using namespace std;

const int N = 1e5 + 10;
struct TreeNode {
  int l, r, f;
} tree[N];

void pre(int x) {
  printf("%d ", x);
  if (tree[x].l) {
    pre(tree[x].l);
  }
  if (tree[x].r) {
    pre(tree[x].r);
  }
}
int main() {
  int n;
  cin >> n;
  for (int i = 1; i <= n; i++) {
    int x, y;
    scanf("%d%d", &x, &y);
    if (x) {
      tree[i].l = x;
      tree[x].f = i;
    }
    if (y) {
      tree[i].r = y;
      tree[x].f = i;
    }
  }
  pre(1);
  printf("\n");
  return 0;
}

    遍历二叉树(链表方式)

    /*
        遍历二叉树, 输入n组关系
        第i组关系表示第i个节点的左孩子是几号节点，右孩子是几号节点。
        为0表示没有节点
*/

#include <algorithm>
#include <iostream>

using namespace std;

struct TreeNode {
  int val;
  TreeNode *l, *r, *f;
} a[10000];

void pre(TreeNode *p) {
  cout << p->val << " ";
  if (p->l) {
    pre(p->l);
  }
  if (p->r) {
    pre(p->r);
  }
}
int main() {
  int n;
  cin >> n;
  for (int i = 1; i <= n; i++) {
    a[i].val = i;

    int x, y;
    cin >> x >> y;
    if (x) {
      a[i].l = &a[x];
      a[x].f = &a[i];
    }
    if (y) {
      a[i].r = &a[y];
      a[y].f = &a[i];
    }
  }
  pre(&a[1]);
  printf("\n");
  return 0;
}
