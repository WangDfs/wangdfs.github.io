并查集

int fa[N]; //父亲

int find(int x){
    if(x == fa[x]){
        return x;
    }
    return fa[x] = find(fa[x]);
}

void union(int x, int y){
    int fx = find(x);
    int fy = find(y);
    
    if(fx == fy){
 		return;       
    }
    fa[fx] = fy;
}
