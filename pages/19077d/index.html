石子游戏

设有 N 堆石子排成一排，其编号为 1,2,3,…,N
每堆石子有一定的质量，可以用一个整数来描述，现在要将这 N堆石子合并成为一堆。
每次只能合并相邻的两堆，合并的代价为这两堆石子的质量之和
例如有 4堆石子分别为 1 3 5 2， 我们可以先合并 1、2 堆，代价为 4，得到 4 5 2， 又合并 1、2 堆，代价为 9，得到 9 2 ，再合并得到 11，总代价为 4+9+11=24；
如果第二步是先合并 2、3堆，则代价为 7，得到 4 7，最后一次合并代价为 11，总代价为 4+7+11=22
问题是：找出一种合理的方法，使总的代价最小，输出最小代价。

可以把第i个位置当成分界点， 左边的最小值，和右边的最小值。然后划分为子问题递归求解，或者dp求解

#include<iostream>
#include<algorithm>
#include<bits/stdc++.h>
using namespace std;
int n, a[310], s[310], f[310][310];
int solve(int l, int r){
    if(f[l][r] != -1){
        return f[l][r];
    }
    if(l == r){
        return f[l][r] = 0;
    }
    int res = 1 << 30;
    for(int i = l; i < r; i++){
       res = min(res, solve(l, i) + solve(i + 1, r));
    }
    return f[l][r] = res + s[r] - s[l - 1];
}
int main(){

    cin >> n;
    for(int i = 1; i <= n; i++){
        cin >> a[i];
    }
    for(int i = 1; i <= n; i++){
        s[i] = s[i - 1] + a[i];
    }
   //cout << solve(1, n);
    
    
   //dp写法
    memset(f, 127, sizeof(f));
    for(int i = 1; i <= n; i++){
        f[i][i] = 0;
    }
    for(int i = 1; i < n; i++){ //区间长度
        for(int j = 1; j <= n - i; j++){ //区间左端点
            for(int k = j; k < j + i; k++){ //遍历j 到 j + i每个点。
                f[j][j + i] = min(f[j][j + i], f[j][k] + f[k + 1][j + i] + s[j + i] - s[j - 1]);
            }
        }
    }
    cout << f[1][n];
    return 0;
}
