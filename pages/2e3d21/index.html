private TrieNode root = new TrieNode();

//表示形式
public static class TrieNode{
    public int pass;
    public int end;
  	//也可以用哈希表来表示
    public TrieNode[] next;
    
    //public HashMap<int, TrieNode> next;
    
    public TrieNode(){
        pass = 0;
        end = 0;
        next = new TrieNode[26];
    }   
}

//插入一个字符串
public void insert(String s){
    
    TrieNode cur = root;
    for(int i = 0; i < s.length(); i++){
        int c = s.charAt(i) - 'a';
        if(cur.next[c] == null){
            cur.next[c] = new TrieNode();
        }
        cur = cur.next[c];
        cur.pass++;
    }
    cur.end++;
}

//查询一个字符串

public boolean query(String s){
    TrieNode cur = root;
    for(int i = 0; i < s.length(); i++){
        int c = s.charAt(i) - 'a';
        if(cur.next[c] == null){
            return false;
        }
        cur = cur.next[c];
        
    }
   return true;
}
