字典树的题

1233. 删除子文件夹 - 力扣（LeetCode） (opens new window)

class Solution {
    private TreeNode root;
    public static class TreeNode{
        public int pass;
        public int end;
        //public TreeNode[] next;
        public HashMap<String, TreeNode> next;

        public TreeNode(){
            pass = 0;
            end = 0;
            next = new HashMap<String, TreeNode>();
        }
    }
    public List<String> removeSubfolders(String[] folder) {
        root = new TreeNode();
        List<String> list = new ArrayList<>();
        TreeNode cur = root;
        for(String s : folder){
            insert(s);
        }
        for(String s : folder){
            if(query(s)){
                list.add(s);
            }
        }
        return list;
    }
    public void insert(String s){
        TreeNode cur = root;
        String str = "";
        for(int i = 1; i < s.length(); i++){
            if(s.charAt(i) == '/'){
                if(!cur.next.containsKey(str)){
                    cur.next.put(str,new TreeNode());
                }
                cur = cur.next.get(str);
                cur.pass++;
                str = "";
            }
            else{
                str += s.charAt(i);
            }
        }
        if(!cur.next.containsKey(str)){
            cur.next.put(str,new TreeNode());
        }
        cur = cur.next.get(str);
        cur.pass++;
        cur.end++;
    }

    public boolean query(String s){
        TreeNode cur = root;
        String str = "";
        for(int i = 1; i < s.length(); i++){
            if(s.charAt(i) == '/'){
                cur = cur.next.get(str);
                if(cur.end > 0 && i != s.length() - 1)return false;
                str = "";
            }
            else{
                str += s.charAt(i);
            }
        }
        if(cur.end > 0)return false;
        cur = cur.next.get(str);
        return true;
    }
}
